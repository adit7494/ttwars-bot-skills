    var origXHRSend = XMLHttpRequest.prototype.send;

    XMLHttpRequest.prototype.open = function(method, url) {
        this._farmbotUrl = url;
        return origXHROpen.apply(this, arguments);
    };

    XMLHttpRequest.prototype.send = function() {
        var self = this;
        if (self._farmbotUrl && self._farmbotUrl.indexOf('/api/v1/farm-list/send') !== -1) {
            var origOnLoad = self.onload;
            var origOnReady = self.onreadystatechange;

            function checkResponse() {
                if (self.readyState === 4 && self.status === 200) {
                    try {
                        var data = JSON.parse(self.responseText);
                        if (data && data.captchaRequired === true) {
                            showCaptcha();
                        }
                    } catch(e) {}
                }
            }

            if (origOnLoad) {
                self.onload = function() {
                    checkResponse();
                    return origOnLoad.apply(this, arguments);
                };
            }

            self.addEventListener('load', checkResponse);
        }
        return origXHRSend.apply(this, arguments);
    };

    var origFetch = window.fetch;
    if (origFetch) {
        window.fetch = function(url, opts) {
            var promise = origFetch.apply(this, arguments);
            if (typeof url === 'string' && url.indexOf('/api/v1/farm-list/send') !== -1) {
                promise.then(function(resp) {
                    return resp.clone().json().then(function(data) {
                        if (data && data.captchaRequired === true) {
                            showCaptcha();
                        }
                    }).catch(function(){});
                }).catch(function(){});
            }
            return promise;
        };
    }
})();
